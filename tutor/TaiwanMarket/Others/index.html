    "end_date": "2020-04-03"
}

data = data.json()

    library(httr)
library(data.table)
library(dplyr)

url = 'https://api.finmindtrade.com/api/v3/data'
response = httr::GET(url = url,
                    query = list(
                    dataset="TaiwanStockNews",
                    data_id="2330",
                    date= "2020-04-01",
                    end_date='2020-04-03'
                    )
)
data = response %>% content
df = do.call('cbind',data$data) %>%data.table
head(df)
